lucas(1,[2]).

lucas(2,[2,1]).

lucas(N,List) :-
	N > 2,
	N1 is N-1,
	N2 is N-2,
	lucas(N1, List1),
	lucas(N2, List2),
	lastElement(List1, F1),
	lastElement(List2, F2),
	F is F1 + F2,
	append(List1, [F], List).

lastElement([Head], List):-
	List = Head.
lastElement([_|Tail], List):-
	lastElement(Tail, List).
